let data = {
 a: 1,
 b: 2,
};
let rule = {
 a: 'number|require|length,0,1',
 b: 'chinese',
};
let name = {
 a: '提示信息1'
 b: '提示信息2'
};
let bool = aui.file.validate(data, rule, name);

键 类型 默认 注释
data object * 需要验证的数据
rule object * 验证器
name object 自定义提示信息
